```c
int main() {
    int i = 42; // 42 est une constante littérale
    int u = 0x2a;
    int o = 052;
    int c = '*'; 
    int b = 0b00101010; // Pas standard selon ISO8899:2011 (standard C)

    printf("%d, %d, %d, %d, %d\n", i, u, o, c, b);
}
```
